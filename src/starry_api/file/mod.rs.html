105#[allow(dead_code)]
106pub trait FileLike: Send + Sync {
107    fn read(&self, buf: &mut [u8]) -> LinuxResult<usize>;
108    fn write(&self, buf: &[u8]) -> LinuxResult<usize>;
109    fn read_at(&self, offset: u64, buf: &mut [u8]) -> LinuxResult<usize> {
110        let _ = offset;
111        let _ = buf;
112        Err(LinuxError::ESPIPE)
113    }
114    fn write_at(&self, offset: u64, buf: &[u8]) -> LinuxResult<usize> {
115        let _ = offset;
116        let _ = buf;
117        Err(LinuxError::ESPIPE)
118    }
119    fn stat(&self) -> LinuxResult<Kstat>;
120    fn truncate(&self, len: u64) -> LinuxResult {
121        let _ = len;
122        Err(LinuxError::EINVAL)
123    }
124    fn fsync(&self) -> LinuxResult {
125        Err(LinuxError::EINVAL)
126    }
127    fn into_any(self: Arc<Self>) -> Arc<dyn Any + Send + Sync>;
128    fn poll(&self) -> LinuxResult<PollState>;
129    fn set_nonblocking(&self, nonblocking: bool) -> LinuxResult;
130
131    fn from_fd(fd: c_int) -> LinuxResult<Arc<Self>>
132    where
133        Self: Sized + 'static,
134    {
135        get_file_like(fd)?
136            .into_any()
137            .downcast::<Self>()
138            .map_err(|_| LinuxError::EINVAL)
139    }
140
141    fn add_to_fd_table(self) -> LinuxResult<c_int>
142    where
143        Self: Sized + 'static,
144    {
145        add_file_like(Arc::new(self))
146    }
147}

149def_resource! {
150    pub static FD_TABLE: ResArc<RwLock<FlattenObjects<Arc<dyn FileLike>, AX_FILE_LIMIT>>> = ResArc::new();
151}
152
153impl FD_TABLE {
154    /// Return a copy of the inner table.
155    pub fn copy_inner(&self) -> RwLock<FlattenObjects<Arc<dyn FileLike>, AX_FILE_LIMIT>> {
156        let table = self.read();
157        let mut new_table = FlattenObjects::new();
158        for id in table.ids() {
159            let _ = new_table.add_at(id, table.get(id).unwrap().clone());
160        }
161        RwLock::new(new_table)
162    }
163
164    pub fn clear(&self) {
165        let mut table = self.write();
166        let ids = table.ids().collect::<Vec<_>>();
167        for id in ids {
168            let _ = table.remove(id);
169        }
170    }
171
172    /// Synchronize all open files in the file descriptor table.
173    pub fn sync_all(&self) -> LinuxResult {
174        let table = self.read();
175        for id in table.ids() {
176            if let Some(file) = table.get(id) {
177                file.fsync()?;
178            }
179        }
180        Ok(())
181    }
182}
183
184/// Get a file-like object by `fd`.
185pub fn get_file_like(fd: c_int) -> LinuxResult<Arc<dyn FileLike>> {
186    FD_TABLE
187        .read()
188        .get(fd as usize)
189        .cloned()
190        .ok_or(LinuxError::EBADF)
191}
192
193/// Add a file to the file descriptor table.
194pub fn add_file_like(f: Arc<dyn FileLike>) -> LinuxResult<c_int> {
195    Ok(FD_TABLE.write().add(f).map_err(|_| LinuxError::EMFILE)? as c_int)
196}
197
198/// Close a file by `fd`.
199pub fn close_file_like(fd: c_int) -> LinuxResult {
200    let f = FD_TABLE
201        .write()
202        .remove(fd as usize)
203        .ok_or(LinuxError::EBADF)?;
204    debug!("close_file_like <= count: {}", Arc::strong_count(&f));
205    Ok(())
206}

208#[ctor_bare::register_ctor]
209fn init_stdio() {
210    let mut fd_table = flatten_objects::FlattenObjects::new();
211    fd_table
212        .add_at(0, Arc::new(stdio::stdin()) as _)
213        .unwrap_or_else(|_| panic!()); // stdin
214    fd_table
215        .add_at(1, Arc::new(stdio::stdout()) as _)
216        .unwrap_or_else(|_| panic!()); // stdout
217    fd_table
218        .add_at(2, Arc::new(stdio::stdout()) as _)
219        .unwrap_or_else(|_| panic!()); // stderr
220    FD_TABLE.init_new(spin::RwLock::new(fd_table));
221}
